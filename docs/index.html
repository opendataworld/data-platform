from agent import DataPlatformAgent

# Create agent with AutonomyX
agent = DataPlatformAgent(model="llama3")
agent.register_with_autonomyx(
    name="data-pipeline-agent",
    sponsor_id="admin-001",
    tenant_id="my-org"
)

# Run orchestration
result = await agent.run(
    "Crawl https://example.com and extract entities"
)

# Quality check
await agent.run(
    "Validate customer_data.csv with Great Expectations"
)
